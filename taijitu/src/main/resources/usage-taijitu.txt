Usage: java taijitu propertiesFile
Example: java taijitu local.properties
See README.MD to an explanation for needed properties, or use the following as base:

database.test.connectionString      = jdbc:h2:mem:test 
database.test.driver                = org.h2.Driver    
database.test.password              = password         
database.test.username              = user             

database.test2.connectionString     = jdbc:h2:mem:test2
database.test2.driver               = org.h2.Driver    
database.test2.password             = password         
database.test2.username             = user             

runtime.test.source.database    = test
runtime.test.source.query       = select * from test
runtime.test.target.database    = test2
runtime.test.target.query       = select * from test2

runtime.test.fields             = KEY,VALUE
runtime.test.fields.compare     = KEY,VALUE
runtime.test.fields.key         = KEY

output.format.xsl = true
